#include<iostream>
#include<math.h>
using namespace std;

class integer
{
    public:
    int n;
void root(int n)
{
     cout<<"Square root is "<<sqrt(n);
}};

int main()
{
     integer inte;
     int n;
     cin>>n;
     inte.root(n);
}
